\section{Evaluation Metric Formulas}

\begin{table}[H]
    \centering
    \caption{Evaluation Metric Formulas}
    \begin{tabular}{|l|l|}
        \hline
        \textbf{Metric} & \textbf{Formula} \\
        \hline
        Accuracy & $(TP + TN) / (TP + TN + FP + FN)$ \\
        \hline
        Precision & $TP / (TP + FP)$ \\
        \hline
        Recall (TPR) & $TP / (TP + FN)$ \\
        \hline
        Specificity (TNR) & $TN / (TN + FP)$ \\
        \hline
        F1-Score & $2 \cdot (P \cdot R) / (P + R)$ \\
        \hline
        MCC & $(TP \cdot TN - FP \cdot FN) / \sqrt{(TP{+}FP)(TP{+}FN)(TN{+}FP)(TN{+}FN)}$ \\
        \hline
        Brier Score & $(1/N) \sum_{i=1}^{N}(p_i - y_i)^2$ \\
        \hline
    \end{tabular}
    \label{tab:metric_formulas}
\end{table}
