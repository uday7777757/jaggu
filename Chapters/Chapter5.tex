\section{Feature Engineering Implementation}

\begin{lstlisting}[language=Python, caption={Feature Engineering Pipeline}, label={lst:feateng}]
def engineer_features(df: pd.DataFrame) -> pd.DataFrame:
    """Derive discriminative features from raw columns."""
    df = df.copy()
    # Drop non-predictive identifiers
    df.drop(columns=['RowNumber', 'CustomerId', 'Surname'],
            errors='ignore', inplace=True)

    # Derived ratio features
    df['BalanceSalaryRatio'] = df['Balance'] / (df['EstimatedSalary'] + 1)
    df['TenureByAge'] = df['Tenure'] / df['Age']
    df['CreditScoreGivenAge'] = df['CreditScore'] / df['Age']
    df['ProductsPerAge'] = df['NumOfProducts'] / df['Age']
    df['BalancePerProduct'] = df['Balance'] / df['NumOfProducts']

    # Binary interaction flags
    df['HasBalance'] = (df['Balance'] > 0).astype(int)
    df['InactiveWithBalance'] = (
        (df['IsActiveMember'] == 0) & (df['Balance'] > 0)
    ).astype(int)
    df['ActiveMultiProduct'] = (
        (df['IsActiveMember'] == 1) & (df['NumOfProducts'] > 1)
    ).astype(int)

    # Age group ordinal encoding
    df['AgeGroup'] = pd.cut(
        df['Age'],
        bins=[17, 30, 45, 60, 100],
        labels=[0, 1, 2, 3]
    ).astype(int)

    # Categorical encoding
    df = pd.get_dummies(df, columns=['Geography'], drop_first=True)
    df['Gender_Male'] = (df['Gender'] == 'Male').astype(int)
    df.drop(columns=['Gender'], inplace=True)

    # Germany-Female interaction (high-risk segment)
    if 'Geography_Germany' in df.columns:
        df['GermanyFemale'] = (
            df['Geography_Germany'].astype(int) * (1 - df['Gender_Male'])
        )
    return df
\end{lstlisting}

from sklearn.pipeline import Pipeline

CONTINUOUS_FEATURES = [
    'CreditScore', 'Age', 'Tenure', 'Balance',

    'ProductsPerAge', 'BalancePerProduct'

def build_preprocessing_pipeline():
    scaler = ColumnTransformer(
        transformers=[
            ('scale', StandardScaler(), CONTINUOUS_FEATURES)
        ],
        remainder='passthrough'
    )
    return scaler

\section{REST API Implementation}

\begin{lstlisting}[language=Python, caption={Flask REST API Endpoint}, label={lst:api}]
from flask import Flask, request, jsonify
import joblib
import shap
import pandas as pd

app = Flask(__name__)

# Load serialised pipeline
pipeline = joblib.load('models/xgb_pipeline.pkl')
explainer = shap.TreeExplainer(pipeline.named_steps['model'])

@app.route('/predict', methods=['POST'])
def predict():
    data = request.get_json(force=True)
    df = pd.DataFrame([data])
    # Feature engineering (same as training)
    df_feat = engineer_features(df)
    # Predict
    prob = pipeline.predict_proba(df_feat)[0][1]
    label = int(prob >= 0.42)   # optimised threshold
    # SHAP explanation
    shap_vals = explainer.shap_values(
        pipeline[:-1].transform(df_feat)
    ).tolist()
    return jsonify({
        'churn_probability': round(float(prob), 4),
        'churn_prediction': label,
        'shap_attributions': dict(
            zip(df_feat.columns.tolist(), shap_vals[0])
        )
    })

@app.route('/health', methods=['GET'])
def health():
    return jsonify({'status': 'ok'})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
\end{lstlisting}

\section{Docker Containerisation}

The following Dockerfile packages the Flask application for deployment:

\begin{lstlisting}[language=,caption={Dockerfile},label={lst:dockerfile}]
FROM python:3.10-slim

WORKDIR /app
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .
EXPOSE 5000
CMD ["gunicorn", "--workers", "4", "--bind", "0.0.0.0:5000", "api:app"]
\end{lstlisting}

The container is built and run using:

\begin{lstlisting}[language=bash, caption={Docker Build and Run Commands}, label={lst:docker}]
docker build -t churn-predictor:latest .

    -v $(pwd)/models:/app/models \
    --name churn-api churn-predictor:latest
\end{lstlisting}
