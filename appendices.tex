\section{API Request and Response Schemas}

\begin{lstlisting}[language=Python, caption={FastAPI Pydantic schemas}]
from pydantic import BaseModel
from typing import List, Dict

class CustomerFeatures(BaseModel):
    customer_id: str
    age: int
    account_tenure_months: int
    monthly_txn_count_30d: float
    monthly_txn_value_30d: float
    days_since_last_txn: int
    login_frequency_30d: float
    credit_utilisation_ratio: float
    support_tickets_90d: int
    product_count: int

class ChurnScore(BaseModel):
    customer_id: str
    churn_probability: float
    churn_flag: bool
    top_shap_features: Dict[str, float]

class BatchScoreResponse(BaseModel):
    model_version: str
    scores: List[ChurnScore]
\end{lstlisting}
